\begin{center}
\includegraphics[scale=.5]{fernbachmodel}
\end{center}

	\textbf{Predictive}\\
	$$P(E\!f\!f\!ect\:|\:Cause) = W_c+W_a-(W_cW_a)$$\\
	
	\textbf{Diagnostic}\\
	$$P(Cause\:|\:E\!f\!f\!ect)=1-P(\:\sim Cause)\frac{P(E\!f\!f\!ect\:|\:\sim Cause)}{P(E\!f\!f\!ect)}$$\\
	$$P(Cause\:|\:E\!f\!f\!ect)=1-(1-P_c)\frac{W_a}{P_cW_c+W_a-P_cW_cW_a}$$
